\documentclass[
  11pt,  % Font size
  a4paper,  % Paper size
]{article}

% Package Imports
\usepackage{amssymb,amsmath}
\usepackage{xcolor}
\usepackage{graphicx}
\usepackage{hyperref}
\usepackage[margin=1in]{geometry}
\usepackage{fancyhdr, lastpage}
\usepackage{setspace}
\usepackage{titlesec}

% Font Setup (fallback to lmodern if no specific fonts are provided)
\usepackage{lmodern}

% Optional Page Number Suppression
$if(suppress_page_numbers)$
  \pagestyle{empty} % Suppress page numbers globally if suppress_page_numbers is set
$else$
  \pagestyle{fancy} % Use custom page style for non-suppressed pages
$endif$

% Fancy Header and Footer Setup
\fancyhf{} % Clear header and footer
\lhead{}
\chead{}
\rhead{}
\lfoot{}
\cfoot{}
\rfoot{\thepage} % Display page number in the right footer

% Title Formatting
\titleformat*{\section}{\Large\sffamily\bfseries\color{black}}
\titleformat*{\subsection}{\large\sffamily\bfseries\color{black}}
\titleformat*{\subsubsection}{\normalsize\sffamily\bfseries\itshape\color{black}}

% Document Title and Author
$if(title)$
\title{$title$}
$endif$
$if(author)$
\author{$author$}
$endif$
$if(date)$
\date{$date$}
$else$
\date{\today}
$endif$

% Begin Document
\begin{document}

\maketitle

% Conditional Author Information (replace with empty if not provided)
$if(author)$
\begin{flushright}
{\large \sffamily \bfseries $author$}\\[.35ex]
$if(address)$
\emph{\small $address$}\\[.35ex]
$endif$
$if(phone)$
\small{$phone$}\\
$endif$
$if(email)$
\small{\href{mailto:$email$}{$email$}}\\
$endif$
$if(url)$
\small{\href{http://$url$}{$url$}}
$endif$
\end{flushright}
$endif$

\vspace{1em}

$body$

% Bibliography Section (if any)
$if(natbib)$
\bibliographystyle{$biblio-style$}
\bibliography{$bibliography$}
$endif$

$if(biblatex)$
\printbibliography
$endif$

\end{document}
